Empty an array

    var a = [0,1,2];
    console.log(a.length) //3
    
    a = [];
    console.log(a.length) //0 empty
    console.log(a)// []
    
    //again
    a= [1,2,3];
    console.log(a.length) //3
    
    a.length = 0;
    console.log(a.length) //0
    
    /*
    you can empty an array by setting the length property 0;
    */

Decrease array index

    var a = [0,1,2];
    console.log(a.length) //3
    
    a.length = 2;
    console.log(a.length) //2
    console.log(a)// 0,1
    
    /*
    It will remove the elements from last
    */

Increase array index

    var a = [0,1,2];
    console.log(a.length) //3
    
    a.length = 10;
    console.log(a.length) //10
    console.log(a)// 0,1,2

Increase array index

    var a = [0,1,2];
    console.log(a.length) //3
    
    a[5] = 5;    
    console.log(a.length) //6
