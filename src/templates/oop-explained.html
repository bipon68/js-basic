    function animal() {
    this.name = 'animal';    
    this.sayAnimal = function () {
        console.log('this is animal');
        }
    }

    function cat() {
        animal.call(this);
        this.name = 'cat';  
        this.sayCat = function () {
            console.log('this is cat');
        }  
    }

    function child() {
        cat.call(this);
        this.name = 'child';
        this.sayChild = function () {
            console.log('this is child');
        }
    }
